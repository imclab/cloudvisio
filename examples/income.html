Per capita income

countries split in high/medium/low groups

Data from worldbank.org

	vis = new Cloudvisio();

	d3.json("data/income.json", function( data ){

		vis.data( data[1] );

		vis.set({ layout: "force" });
		vis.axis("radius", 5);

		vis.axis("name");
		vis.group("incomeLevel", ["High", "Medium", "Low"]);

		vis.render();
	});
